package com.example.invisiblegardening.models;

import com.fasterxml.jackson.annotation.JsonBackReference;

import javax.persistence.*;
import java.util.List;

@Entity
public class CustomerData {

    @Id
    @GeneratedValue
    Long id;

    String name;
    String address;
    String zipcode;
    String city;
    String emailaddress;
    String phoneNumber;
    Boolean selectCompany;

    @OneToMany(mappedBy = "customerData")
    List<CustomerRequest> customerRequests;

    @OneToOne
    @JsonBackReference("customerDataCompany")
    Company company;

    public Long getId() {
        return id;
    }

    public String getName() {
        return name;
    }

    public String getAddress() {
        return address;
    }

    public String getZipcode() {
        return zipcode;
    }

    public String getCity() {
        return city;
    }

    public String getEmailaddress() {
        return emailaddress;
    }

    public String getPhoneNumber() {
        return phoneNumber;
    }

    public Boolean isSelectCompany() {
        return selectCompany;
    }

    public Company getCompany() {
        return company;
    }

    public Boolean getSelectCompany() {
        return selectCompany;
    }

    public List<CustomerRequest> getCustomerRequests() {
        return customerRequests;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public void setName(String name) {
        this.name = name;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public void setZipcode(String zipcode) {
        this.zipcode = zipcode;
    }

    public void setCity(String city) {
        this.city = city;
    }

    public void setEmailaddress(String emailaddress) {
        this.emailaddress = emailaddress;
    }

    public void setPhoneNumber(String phoneNumber) {
        this.phoneNumber = phoneNumber;
    }

    public void setSelectCompany(Boolean company) {
        this.selectCompany = company;
    }

    public void setCompany(Company company) {
        this.company = company;
    }

    public void setCustomerRequests(List<CustomerRequest> customerRequests) {
        this.customerRequests = customerRequests;
    }
}

package com.example.invisiblegardening.models;

import javax.persistence.*;
import java.util.Date;
import java.util.List;

@Entity
public class Machine {

    @Id
    @GeneratedValue
    Long id;

    String name;
    String description;
    String type;
    String measurements;
    Double valueOfPurchase;
    String status;
    Date lastService;
    Date plannedService;

    @Lob
    byte[] image;

    @ManyToMany(mappedBy = "machines")
    List<Job> jobs;

    @ManyToMany(mappedBy = "machines")
    List<CustomerRequest> customerRequests;

    public Long getId() {
        return id;
    }

    public String getName() {
        return name;
    }

    public String getDescription() {
        return description;
    }

    public String getType() {
        return type;
    }

    public String getMeasurements() {
        return measurements;
    }

    public Double getValueOfPurchase() {
        return valueOfPurchase;
    }

    public String getStatus() {
        return status;
    }

    public Date getLastService() {
        return lastService;
    }

    public Date getPlannedService() {
        return plannedService;
    }

    public byte[] getImage() {
        return image;
    }

    public List<Job> getJobs() {
        return jobs;
    }

    public List<CustomerRequest> getCustomerRequests() {
        return customerRequests;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public void setName(String name) {
        this.name = name;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public void setType(String type) {
        this.type = type;
    }

    public void setMeasurements(String measurements) {
        this.measurements = measurements;
    }

    public void setValueOfPurchase(Double valueOfPurchase) {
        this.valueOfPurchase = valueOfPurchase;
    }

    public void setStatus(String status) {
        this.status = status;
    }

    public void setLastService(Date lastService) {
        this.lastService = lastService;
    }

    public void setPlannedService(Date plannedService) {
        this.plannedService = plannedService;
    }

    public void setImage(byte[] image) {
        this.image = image;
    }

    public void setJobs(List<Job> jobs) {
        this.jobs = jobs;
    }

    public void setCustomerRequests(List<CustomerRequest> customerRequests) {
        this.customerRequests = customerRequests;
    }
}
